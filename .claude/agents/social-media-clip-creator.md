process_for_all_platforms() {
    local input_video="$1"
    local base_name="${input_video%.*}"
    
    # TikTok/Reels optimization (9:16, engagement-focused)
    ffmpeg -i "$input_video" \
        -vf "scale=1080:1920:force_original_aspect_ratio=increase,crop=1080:1920,subtitles=captions.srt:force_style='FontSize=24,PrimaryColour=&Hffffff,OutlineColour=&H000000'" \

        -c:v libx264 -preset medium -crf 20 \

        -movflags +faststart \
