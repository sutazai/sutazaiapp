pythonclass ModelPredictiveController:
    def __init__(self, system_model, constraints, horizon):
        self.system_model = system_model
        self.constraints = constraints
        self.prediction_horizon = horizon
        self.control_horizon = horizon // 2
        
    def compute_optimal_control(self, current_state, reference_trajectory):
        """Solve MPC optimization problem"""
        # Define optimization variables
        u = cp.Variable((self.control_horizon, self.system_model.input_dim))
        x = cp.Variable((self.prediction_horizon + 1, self.system_model.state_dim))
        
        # Define constraints
        constraints = [x[0] == current_state]
        
        for k in range(self.prediction_horizon):
            # System dynamics constraint
            constraints.append(x[k+1] == self.system_model.predict(x[k], u[min(k, self.control_horizon-1)]))
            
            # State and input constraints
            constraints.append(self.constraints.state_bounds[0] <= x[k] <= self.constraints.state_bounds[1])
            if k < self.control_horizon:
                constraints.append(self.constraints.input_bounds[0] <= u[k] <= self.constraints.input_bounds[1])
        
        # Define objective function
        cost = 0
        for k in range(self.prediction_horizon):
            cost += cp.quad_form(x[k] - reference_trajectory[k], self.constraints.Q)
            if k < self.control_horizon:
                cost += cp.quad_form(u[k], self.constraints.R)
        
        # Solve optimization problem
        problem = cp.Problem(cp.Minimize(cost), constraints)
        problem.solve()
        
        return u[0].value if problem.status == cp.OPTIMAL else None

pythonclass ExtendedKalmanFilter:
    def __init__(self, state_dim, measurement_dim):
        self.state_dim = state_dim
        self.measurement_dim = measurement_dim
        self.state_estimate = np.zeros(state_dim)
        self.covariance_matrix = np.eye(state_dim)
        
    def predict(self, control_input, process_noise):
        """Prediction step of EKF"""
        # Predict state using motion model
        predicted_state = self.motion_model(self.state_estimate, control_input)
        
        # Compute Jacobian of motion model
        F = self.compute_motion_jacobian(self.state_estimate, control_input)
        
        # Predict covariance
        predicted_covariance = F @ self.covariance_matrix @ F.T + process_noise
        
        self.state_estimate = predicted_state
        self.covariance_matrix = predicted_covariance
        
    def update(self, measurement, measurement_noise):
        """Update step of EKF"""
        # Predict measurement
        predicted_measurement = self.measurement_model(self.state_estimate)
        
        # Compute Jacobian of measurement model
        H = self.compute_measurement_jacobian(self.state_estimate)
        
        # Compute Kalman gain
        S = H @ self.covariance_matrix @ H.T + measurement_noise
        K = self.covariance_matrix @ H.T @ np.linalg.inv(S)
        
        # Update state and covariance
        measurement_residual = measurement - predicted_measurement
        self.state_estimate += K @ measurement_residual
        self.covariance_matrix = (np.eye(self.state_dim) - K @ H) @ self.covariance_matrix
