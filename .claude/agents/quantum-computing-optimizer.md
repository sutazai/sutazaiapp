                qml.CNOT(wires=[i, i + 1])

                qml.CNOT(wires=[i, i + 1])

        return [qml.expval(qml.PauliZ(i)) for i in range(self.n_qubits)]

    ) -> np.ndarray:

        n = len(agents)
        Q = np.zeros((n * n, n * n))
        
        # Encode agent interactions
        for i in range(n):
            for j in range(n):

                    # Coupling strength based on interaction benefit
