### Agent Health Monitoring

```bash
#!/bin/bash
# Real script: /opt/sutazaiapp/scripts/monitoring/agent_health_check.sh

perform_agent_health_check() {
    local agent_registry="/opt/sutazaiapp/config/agent_registry.json"
    local health_report="/var/log/sutazai/agent_health_$(date +%Y%m%d_%H%M%S).log"
    
    log_info "Starting comprehensive agent health check"
    
    # Check each registered agent
    while read -r agent_config; do
        agent_id=$(echo "$agent_config" | jq -r '.id')
        agent_port=$(echo "$agent_config" | jq -r '.port')
        agent_status=$(check_agent_status "$agent_id" "$agent_port")
        
        echo "$(date -u '+%Y-%m-%d %H:%M:%S UTC') - Agent: $agent_id - Status: $agent_status" >> "$health_report"
        
        if [[ "$agent_status" != "HEALTHY" ]]; then
            escalate_agent_issue "$agent_id" "$agent_status"
        fi
        
    done < <(jq -c '.agents[]' "$agent_registry")
    
    log_info "Agent health check completed - report: $health_report"
}

check_agent_status() {
    local agent_id="$1"
    local agent_port="$2"
    
    # Health check implementation
    if curl -s -f "http://localhost:${agent_port}/health" > /dev/null; then
        echo "HEALTHY"
    else
        echo "UNHEALTHY"
    fi
}
```

### Deployment Automation

```bash
#!/bin/bash
# Real script: /opt/sutazaiapp/scripts/deployment/deploy_agent.sh

deploy_agent() {
    local agent_name="$1"
    local agent_version="$2"
    local deployment_environment="$3"
    
    log_info "Deploying agent: $agent_name v$agent_version to $deployment_environment"
    
    # Pre-deployment validation
    validate_agent_package "$agent_name" "$agent_version" || {
        log_error "Agent package validation failed"
        return 1
    }
    
    # Backup current version
    backup_current_agent "$agent_name" || {
        log_error "Failed to backup current agent"
        return 1
    }
    
    # Deploy new version
    deploy_agent_package "$agent_name" "$agent_version" "$deployment_environment" || {
        log_error "Agent deployment failed"
        rollback_agent_deployment "$agent_name"
        return 1
    }
    
    # Post-deployment validation
    validate_agent_deployment "$agent_name" || {
        log_error "Post-deployment validation failed"
        rollback_agent_deployment "$agent_name"
        return 1
    }
    
    # Update agent registry
    update_agent_registry "$agent_name" "$agent_version" "$deployment_environment"
    
    log_success "Agent deployment completed successfully"
    return 0
}
```
